🐍 User: 帮我写一个猜数字游戏
import random
 
# 生成1-100的随机数
answer = random.randint(1, 100)
 
for i in range(10):
guess = int(input("猜一个1-100的数字: "))
if guess == answer:
print("🎉 猜对了！答案就是", answer)
break
elif guess < answer:
print("太小了！")
else:
print("太大了！")
print("游戏结束！")
